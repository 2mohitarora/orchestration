# This is one time setup so that Consul cam use TLS to verify the authenticity of servers and clients.

# Initiatize the buily in CA
consul tls ca create

# Create server certificates (-node flag)
consul tls cert create -server -dc east -additional-dnsname="*.west.consul" -additional-dnsname="*.east.consul" -additional-dnsname="*.server.east.consul" -additional-dnsname="*.server.west.consul"
